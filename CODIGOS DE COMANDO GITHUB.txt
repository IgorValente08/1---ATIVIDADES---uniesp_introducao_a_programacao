git init (iniciar um respositório)

git status (verificar atualização)

git add "nome do arquivo.txt" (adicionar arquivo no repositório)

git add . (adicionar todos os aqruivos no respositório)

git commit -m "commit inicial" (criar versões dos códigos)

git conf --global user.email "" 

git remote add origin 

git push --set-upstream origin master

git reflog

git reset --hard
